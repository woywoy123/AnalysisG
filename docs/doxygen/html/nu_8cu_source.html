118 met_xy -> packed_accessor64<scalar_t, 2, torch::RestrictPtrTraits>(),
119 H -> packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
120 X.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
121 M.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
122 Unit.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>());
123
124 });
125 if (no_sig){delete sigma;}
126
127 torch::Tensor M_ = M.clone();
128 std::map<std::string, torch::Tensor> out = nusol_::Intersection(&M_, &Unit, null);
129
130 const dim3 thN = dim3(1, 6, 3);
131 const dim3 blN = blk_(dx, 1, 6, 6, 3, 3);
132 torch::Tensor nu = torch::zeros({dx, 6, 3}, MakeOp(H));
133 torch::Tensor chi2 = torch::zeros({dx, 6, 1}, MakeOp(H));
134 AT_DISPATCH_ALL_TYPES(H -> scalar_type(), "Nu", [&]{
135 _chi2<scalar_t><<<blN, thN>>>(
136 out["solutions"].packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
137 out["distances"].packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
138 H -> packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
139 X.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
140 nu.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>(),
141 chi2.packed_accessor64<scalar_t, 3, torch::RestrictPtrTraits>());
142 });
143 out["X"] = X;
144 out["M"] = M;
145 out["nu"] = nu;
146 out["distances"] = chi2.view({-1, 6});
