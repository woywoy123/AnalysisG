60  """Linear factors of degenerate quadratic polynomial"""
61  if G[0, 0] == 0 == G[1, 1]: return [[G[0, 1], 0, G[1, 2]], [0, G[0, 1], G[0, 2] - G[1, 2]]]

64  Q /= Q[1, 1]
