25void analysis::build_model_session(){
26
27 if (!this -> model_sessions.size()){return this -> info("No Models Specified. Skipping.");}
28
29 // Configure k-fold cross-validation
30 std::vector<int> kfold = this -> m_settings.kfold;
31 if (!kfold.size()){
32 // If no specific folds are specified, create all folds from 0 to kfolds-1
33 for (int k(0); k < this -> m_settings.kfolds; ++k){
34 kfold.push_back(k);
35 }
36 }
37 else {
38 // Adjust fold indices (users typically specify folds starting from 1)
39 for (size_t k(0); k < kfold.size(); ++k){
40 kfold[k] = kfold[k]-1;
41 }
42 }
43 this -> m_settings.kfold = kfold;
44
45 // Set up device maps for tensor operations
46 std::map<int, torch::TensorOptions*> dev_map;
47 for (size_t x(0); x < this -> model_sessions.size(); ++x){
48 // Extract device options from the model
49 torch::TensorOptions* op = std::get<0>(this -> model_sessions[x]) -> m_option;
50 int dx = op -> device().index();
51
52 // Only add each device once to the map
53 if (dev_map.count(dx)){continue;}
54 dev_map[dx] = op;
55 }
56
57 // Transfer data graphs to appropriate devices
58 this -> loader -> datatransfer(&dev_map);
59
60 // Create and configure optimizer for each model session
61 for (size_t x(0); x < this -> model_sessions.size(); ++x){
62 std::string name = this -> model_session_names[x];
63
64 // Initialize optimizer with appropriate settings
65 optimizer* optim = new optimizer();
66 optim -> m_settings = this -> m_settings;
67 optim -> m_settings.run_name = name;
68 optim -> import_dataloader(this -> loader);
69 this -> trainer[name] = optim;
70
71 // Import model session parameters
72 std::tuple<model_template*, optimizer_params_t*> para = this -> model_sessions[x];
73 optim -> import_model_sessions(&para);
74
75 // Configure training data for each k-fold
76 for (size_t k(0); k < this -> m_settings.kfold.size(); ++k){
77 int k_ = this -> m_settings.kfold[k];
78 std::vector<graph_t*>* check = this -> loader -> get_k_train_set(k_);
79 if (!check){continue;}
80 model_report* mx = nullptr;
81 if (this -> m_settings.debug_mode){initialize_loop(optim, k_, std::get<0>(para), std::get<1>(para), &mx);}
82 else {this -> threads.push_back(new std::thread(initialize_loop, optim, k_, std::get<0>(para), std::get<1>(para), &mx));}
83 while (!mx){std::this_thread::sleep_for(std::chrono::microseconds(10));}
84 this -> reports[mx -> run_name + std::to_string(mx -> k)] = mx;
85 }
