61 double pt_s_ = 0, pt_e_ = 0;
62 for (int i(0); i < n_pt; ++i){
63 pt_s_ = this -> pt_start + i * this -> pt_step;
64 pt_e_ = this -> pt_start + (i+1) * this -> pt_step;
65 if (pt_s_ < pt_p && pt_p < pt_e_){break;}
66 }
67
68 double eta_s_ = 0, eta_e_ = 0;
69 for (int i(0); i < n_eta; ++i){
70 eta_s_ = this -> eta_start + i * this -> eta_step;
71 eta_e_ = this -> eta_start + (i+1) * this -> eta_step;
72 if (eta_s_ < eta_p && eta_p < eta_e_){break;}
73 }
74
75 std::string key = "";
76 key += this -> to_string(pt_s_, 0) + " < $pt_{top}$ < " + this -> to_string(pt_e_, 0) + ", ";
77 key += this -> to_string(eta_s_, 1) + " < |$\\eta_{top}$| < " + this -> to_string(eta_e_, 1);
78 return key;

95bool topefficiency::strategy(event_template* ev){
96 gnn_event* evn = (gnn_event*)ev;
97 std::string hash = evn -> hash;
98
99 // ---------------- Truth Section ---------------- //
100 std::map<std::string, bool> t_top_map;
101 std::vector<top*> truth_tops = evn -> t_tops;
102 std::string decay_region_t = this -> decaymode(truth_tops);
103 for (size_t x(0); x < truth_tops.size(); ++x){
104 top* top_ = truth_tops[x];
105 float mass = top_ -> mass / 1000;
106 std::string key = this -> region(top_ -> pt / 1000, std::abs(top_ -> eta));
107 this -> t_topmass[key][hash].push_back(mass);
108 t_top_map[top_ -> hash] = false;
109 this -> t_nodes[hash][mass] = top_ -> n_nodes;
110 this -> t_decay_region[decay_region_t][hash].push_back(mass);
111 }
112 this -> n_tru_tops[hash] = t_top_map.size();
113
114 std::vector<zprime*> truth_zprime = evn -> t_zprime;
115 for (size_t x(0); x < truth_zprime.size(); ++x){
116 zprime* zp_ = truth_zprime[x];
117 std::string key = this -> region(zp_ -> pt / 1000, std::abs(zp_ -> eta));
118 this -> t_zmass[key][hash].push_back(zp_ -> mass / 1000);
119 }
120
121 // ----------------- Reconstructed Section ------------------- //
122 std::vector<zprime*> reco_zprime = evn -> r_zprime;
123 for (size_t x(0); x < reco_zprime.size(); ++x){
124 zprime* zp_ = reco_zprime[x];
125 std::string key = this -> region(zp_ -> pt / 1000, std::abs(zp_ -> eta));
126 this -> p_zmass[key][hash].push_back(zp_ -> mass / 1000);
127 this -> prob_zprime[key][hash].push_back(zp_ -> av_score);
128 }
129
130 std::vector<top*> reco_tops = evn -> r_tops;
131 std::string decay_region_p = this -> decaymode(reco_tops);
132 for (size_t x(0); x < reco_tops.size(); ++x){
133 top* top_ = reco_tops[x];
134 float mass = top_ -> mass / 1000;
135 std::string key = this -> region(top_ -> pt / 1000, std::abs(top_ -> eta));
136 this -> p_topmass[key][hash].push_back(mass);
137 this -> prob_tops[key][hash].push_back(top_ -> av_score);
138 this -> p_nodes[hash][mass] = top_ -> n_nodes;
139 this -> p_decay_region[decay_region_p][hash].push_back(mass);
140 }
141
142 // ------------------ Efficiency Reconstruction ------------------- //
143 for (size_t s(0); s <= int(1.0/this -> score_step); ++s){
144 int perf = 0;
145 int reco = 0;
146 float sc = s*this -> score_step;
147 for (size_t x(0); x < reco_tops.size(); ++x){
148 top* top_ = reco_tops[x];
149 std::string id = top_ -> hash;
150 if (top_ -> av_score <= sc){continue;}
151 ++reco;
152
153 if (!t_top_map.count(id)){continue;}
154 if (t_top_map[id]){continue;} // prevent double counting
155 t_top_map[id] = true;
156 ++perf;
157 }
158 this -> n_perfect_tops[hash][sc] = perf;
159 this -> n_pred_tops[hash][sc] = reco;
160 std::map<std::string, bool>::iterator ib = t_top_map.begin();
161 for (; ib != t_top_map.end(); ++ib){ib -> second = false;}
162 }
163
164 this -> truth_top_edge = evn -> t_edge_top;
165 this -> pred_top_edge_score = {evn -> edge_top_scores};
166
167 this -> truth_res_edge = evn -> t_edge_res;
168 this -> pred_res_edge_score = {evn -> edge_res_scores};
169
170 this -> truth_signal = std::vector<int>({evn -> t_signal});
171 this -> pred_signal_score = {evn -> signal_scores};
172
173 this -> truth_ntops = std::vector<int>({evn -> t_ntops});
174 this -> pred_ntops_score = {evn -> ntops_scores};
175 return true;
176}
