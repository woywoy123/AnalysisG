142 auto g_data = [this](graph_t* d, std::string key) -> torch::Tensor* {return d -> get_data_graph(key, this);};
143 auto n_data = [this](graph_t* d, std::string key) -> torch::Tensor* {return d -> get_data_node(key, this);};
144 auto e_data = [this](graph_t* d, std::string key) -> torch::Tensor* {return d -> get_data_edge(key, this);};

146 auto n_truth = [this](graph_t* d, std::string key) -> torch::Tensor* {return d -> get_truth_node(key, this);};
147 auto e_truth = [this](graph_t* d, std::string key) -> torch::Tensor* {return d -> get_truth_edge(key, this);};
148 auto lamb_d = [this](
149 std::function<torch::Tensor*(graph_t*, std::string)> fx,
150 std::vector<graph_t*>* merge, std::string key) -> torch::Tensor*
151 {
152 std::vector<torch::Tensor> arr;
153 for (size_t x(0); x < merge -> size(); ++x){
154 torch::Tensor* val = fx((*merge)[x], key);
155 if (!val){continue;}
156 arr.push_back(*val);
157 }
158 if (!arr.size()){return nullptr;}
159 return new torch::Tensor(torch::cat(arr, {0}));
160 };

163 switch (type){
164 case graph_enum::data_graph: tn = lamb_d(g_data, data, inpt); this -> m_i_graph[inpt] = tn; return tn;
