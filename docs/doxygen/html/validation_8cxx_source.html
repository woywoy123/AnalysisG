136bool validation::strategy(event_template* ev){

144 packet_t* pkt_tj = this -> build_packet(evn, "truthjet" );

146 packet_t* pkt_jl = this -> build_packet(evn, "jetleptons" );
147
148 std::vector<top*> tops = this -> upcast<top>(&evn -> Tops);
149 for (size_t x(0); x < tops.size(); ++x){
150 this -> update_state(nullptr); // reset the global matching
151
152 top* tpx = tops[x];
153 std::vector<particle_template*> ch = this -> vectorize(&tpx -> children);
154 for (size_t c(0); c < ch.size(); ++c){
155 if (ch[c] -> is_b){this -> b_qrk = ch[c];}
156 if (ch[c] -> is_lep){this -> lepton = ch[c];}
157 if (ch[c] -> is_nu){this -> nu_tru = ch[c];}
158 }
159 this -> update_state(pkt_tc);
160 this -> b_qrk = nullptr;
161

163 truthjet* ptr = tpx -> TruthJets[c];
164 if (!ptr -> is_b){continue;}
165 this -> b_qrk = ptr; break;
166 }
167
168 this -> update_state(pkt_tj);
169 this -> b_qrk = nullptr;
170

172 jet* ptr = tpx -> Jets[c];

174 this -> b_qrk = ptr; break;
