100bool matching::match_obj(
101 std::vector<particle_template*>* vx, std::vector<particle_template*>* out,
102 std::string hash_, std::vector<int>* num_merged, int* num_jets, bool exl_lep
103){
104
105 auto lmb = [](std::map<std::string, double>* in){
106 std::vector<double> out = {};
107 std::map<std::string, double>::iterator itr = in -> begin();
108 for (; itr != in -> end(); ++itr){out.push_back(itr -> second);}
109 return out;
110 };
111
112 *num_jets = 0;
113 out -> clear();
114 num_merged -> clear();
115 bool is_lepx = false;
116 std::map<std::string, double> frac = {};
117 std::map<std::string, particle_template*> tmp = {};
118
119 for (size_t y(0); y < vx -> size(); ++y){
120 particle_template* ptr = (*vx)[y];
121 std::map<std::string, particle_template*> prnt = ptr -> parents;
122 if (!prnt.count(hash_)){continue;}
123 if (exl_lep && ptr -> is_lep){is_lepx = true; continue;}
124 if (ptr -> is_lep){is_lepx = true;}
125 std::string hashx = ptr -> hash;
126 frac[hashx] = ptr -> e / 1000.0;
127 tmp[hashx] = ptr;
128 }
129 std::vector<double> energy_frac = lmb(&frac);
130 std::vector<particle_template*> tmp_v = this -> vectorize(&tmp);
131
132 double tl = this -> tools::sum(&energy_frac);
133 for (size_t y(0); y < energy_frac.size(); ++y){
134 particle_template* tx = tmp_v[y];
135 std::map<std::string, particle_template*> prnt = tx -> parents;
136 if (energy_frac[y] / tl < this -> energy_constraint){continue;}
137 out -> push_back(tx);
138 if (tx -> is_lep){continue;}
139 num_merged -> push_back(int(prnt.size()));
140 *num_jets += 1;
141 }
142 return is_lepx;
143}
