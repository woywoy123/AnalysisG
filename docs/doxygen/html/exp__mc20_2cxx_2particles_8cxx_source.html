136void physics_truth::build(std::map<std::string, particle_template*>* prt, element_t* el){
137 auto lamb = [](physics_truth* msp, std::vector<int>* prt){
138 int pdg = msp -> pdgid;
139 pdg = std::abs(pdg);
140 if (!prt -> size()){return false;}
141 if (pdg == 0){return false;}
142 if (pdg == 6){return false;}
143 if (pdg == 24){return false;}
144 if (pdg >= 11 && pdg <= 16){return false;}
145 if (msp -> top_index.size() > 1){return true;}
146 if (msp -> top_index[0] == -1){return false;}
147
148 int mx(0);
149 std::map<int, int> out;
150 std::map<int, int>::iterator itx;
151 for (size_t x(0); x < prt -> size(); ++x){out[std::abs((*prt)[x])] += 1;}
152 for (itx = out.begin(); itx != out.end(); ++itx){
153 if (itx -> second < mx){continue;}
154 mx = itx -> second; pdg = itx -> first;
155 }
156 msp -> pdgid = pdg;
157 return true;
158 };
