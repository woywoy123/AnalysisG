Basic Usage

from AnalysisG.core import Analysis
# Create analysis instance
ana = Analysis()
# Configure settings
ana.OutputPath = "results/"
ana.Threads = 8
ana.Epochs = 100
ana.kFolds = 5
ana.BatchSize = 64
# Add data samples
ana.AddSamples("/path/to/data/", "sample_label")
# Add templates
ana.AddEvent(MyEvent(), "sample_label")
ana.AddGraph(MyGraph(), "sample_label")
ana.AddSelection(MySelection())
# Add model for training
opt = OptimizerConfig()
opt.lr = 0.001
ana.AddModel(MyModel(), opt, "training_run")
# Start analysis
ana.Start()
