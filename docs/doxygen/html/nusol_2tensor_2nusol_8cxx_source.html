130torch::Tensor Rotation(torch::Tensor* pmc_b, torch::Tensor* pmc_mu, torch::Tensor* base){
131 torch::Tensor pmc_b3 = pmc_b -> index( {torch::indexing::Slice(), torch::indexing::Slice(0, 3)});
132 torch::Tensor pmc_mu3 = pmc_mu -> index({torch::indexing::Slice(), torch::indexing::Slice(0, 3)});
133
134 torch::Tensor muphi = -transform_::Phi(pmc_mu);
135 torch::Tensor theta = physics_::Theta(pmc_mu);
136
137 torch::Tensor Rz = operators_::Rz(&muphi);
138 torch::Tensor rx = Pi_2(&theta) - theta;
139 torch::Tensor Ry = operators_::Ry(&rx);
140
141 torch::Tensor Rx = torch::matmul(Rz, pmc_b3.view({-1, 3, 1}));
142 Rx = torch::matmul(Ry, Rx.view({-1, 3, 1}));
143 Rx = -torch::atan2(
144 Rx.index({torch::indexing::Slice(), 2}),
145 Rx.index({torch::indexing::Slice(), 1})
146 ).view({-1, 1});
147
148 Rx = operators_::Rx(&Rx);
149 Rx = torch::transpose(Rx, 1, 2);
150 Ry = torch::transpose(Ry, 1, 2);
151 Rz = torch::transpose(Rz, 1, 2);
152 return torch::matmul(torch::matmul(Rz, torch::matmul(Ry, Rx)), *base);
153}

198torch::Tensor nusol_::BaseMatrix(torch::Tensor* pmc_b, torch::Tensor* pmc_mu, torch::Tensor* masses){
199
200 torch::Tensor pmx_b = pmc_b -> view({-1, 4});
201 torch::Tensor pmx_mu = pmc_mu -> view({-1, 4});
202 std::map<std::string, torch::Tensor> mass = GetMasses(&pmx_b, masses);
203
204 const torch::TensorOptions op = MakeOp(pmc_b);
205 const unsigned int dim_i = pmx_b.size(0);
206
207 torch::Tensor bB = physics_::Beta(&pmx_b);
208 torch::Tensor muB = physics_::Beta(&pmx_mu);
209 torch::Tensor muB2 = physics_::Beta2(&pmx_mu);
210 torch::Tensor muP = physics_::P(&pmx_mu);
211
212 torch::Tensor m2b = physics_::M2(&pmx_b);
213 torch::Tensor m2l = physics_::M2(&pmx_mu);
214
215 torch::Tensor x0p = _x0(&pmx_b , &m2b, &mass["T2"], &mass["W2"]);
216 torch::Tensor x0 = _x0(&pmx_mu, &m2l, &mass["W2"], &mass["N2"]);
217
218 torch::Tensor pmc_b3 = pmx_b.index( {torch::indexing::Slice(), torch::indexing::Slice(0, 3)});
219 torch::Tensor pmc_mu3 = pmx_mu.index({torch::indexing::Slice(), torch::indexing::Slice(0, 3)});
220
221 torch::Tensor c_ = operators_::CosTheta(&pmc_b3, &pmc_mu3);
222 torch::Tensor s_ = operators_::SinTheta(&pmc_b3, &pmc_mu3);
223
224 torch::Tensor tmp_ = muB / bB;
225 torch::Tensor w_ = ( - tmp_ - c_ ) / s_;
226 torch::Tensor w = ( tmp_ - c_ ) / s_;
227
228 torch::Tensor O2 = w.pow(2) + 1 - muB2;
229 torch::Tensor e2 = (mass["W2"] - mass["N2"]) * ( 1 - muB2 );
230
231 torch::Tensor Sx = (x0 * muB - muP * ( 1 - muB2 )) / muB2;
232 torch::Tensor Sy = ( (x0p / bB) - c_ * Sx ) / s_;
233
234 tmp_ = Sx + w*Sy;
235 torch::Tensor x1 = Sx - tmp_ / O2;
236 torch::Tensor y1 = Sy - tmp_ * (w / O2);
237 torch::Tensor Z = torch::sqrt(torch::relu(x1.pow(2) * O2 - ( Sy - w*Sx ).pow(2) - ( mass["W2"] - x0.pow(2) - e2 )));
238
239 torch::Tensor O = torch::sqrt(O2);
240 torch::Tensor _0 = torch::zeros({dim_i, 1}, op);
241 return torch::cat({ Z / O, _0, x1 - muP, (w*Z)/O, _0, y1, _0, Z, _0}, -1).view({dim_i, 3, 3});
242}

253std::tuple<torch::Tensor, torch::Tensor> nusol_::Intersection(torch::Tensor* A, torch::Tensor* B, double null){
254 const torch::TensorOptions op = MakeOp(A);
255 const unsigned int dim_i = A -> size(0);
256 std::vector<signed long> dim11 = {-1, 1, 1};
257 std::vector<signed long> dim13 = {-1, 1, 3};
258 std::vector<signed long> dim31 = {-1, 3, 1};
259 std::vector<signed long> dim313 = {-1, 3, 1, 3};
260 std::vector<signed long> dim133 = {-1, 1, 3, 3};
261 std::vector<signed long> dim331 = {-1, 3, 3, 1};
262 std::vector<signed long> dim333 = {-1, 3, 3, 3};
263 torch::Tensor _A = A -> clone();
264 torch::Tensor _B = B -> clone();
265
266 torch::Tensor detA = torch::abs(operators_::Determinant(A));
267 torch::Tensor detB = torch::abs(operators_::Determinant(B));
268
269 // Perform the variable swaps
270 torch::Tensor swp = detB > detA;
271 torch::Tensor _tmp = B -> index({swp});
272 _B.index_put_({swp}, A -> index({swp}));
273 _A.index_put_({swp}, _tmp);
274
275 // Find the non imaginary eigenvalue solutions
276 _tmp = torch::linalg::eigvals(operators_::Inverse(&_A).matmul(_B));
277 torch::Tensor _r = torch::real(_tmp);
278 torch::Tensor msk = torch::isreal(_tmp)*torch::arange(3, 0, -1, op);
279 msk = torch::argmax(msk, -1, true);
280 _r = torch::gather(_r, 1, msk).view({-1, 1, 1});
281 torch::Tensor G = _B - _r*_A;
282
283 // Get the diagonals of the matrix
284 torch::Tensor G00 = G.index({torch::indexing::Slice(), 0, 0});
285 torch::Tensor G11 = G.index({torch::indexing::Slice(), 1, 1});
286
287 // ----- Numerical Stability part ----- //
288 swp = torch::abs(G00) > torch::abs(G11);
289 G.index_put_({swp}, torch::cat({
290 G.index({swp, 1, torch::indexing::Slice()}).view(dim13),
291 G.index({swp, 0, torch::indexing::Slice()}).view(dim13),
292 G.index({swp, 2, torch::indexing::Slice()}).view(dim13)
293 }, 1));
294
295 G.index_put_({swp}, torch::cat({
296 G.index({swp, torch::indexing::Slice(), 1}).view(dim31),
297 G.index({swp, torch::indexing::Slice(), 0}).view(dim31),
298 G.index({swp, torch::indexing::Slice(), 2}).view(dim31)
299 }, 2));
300
301 G = G/(G.index({torch::indexing::Slice(), 1, 1}).view({-1, 1, 1}));
302
303 torch::Tensor CoF = operators_::CoFactors(&G);
304 torch::Tensor g22 = CoF.index({torch::indexing::Slice(), 2, 2});
305 torch::Tensor _out = zeros_like(G);
306
307 detA = detA == 0;
308 torch::Tensor G00_G11 = ( G00 == 0 ) * (G11 == 0);
309 torch::Tensor g22__ = (-g22 <= 0.) * (G00_G11 == false);
310 torch::Tensor g22_ = (-g22 > 0.) * (G00_G11 == false);
311
312 // 1. ----- Case where the solutions are Horizontal and Vertical ----- //
313 torch::Tensor tx = G.index({G00_G11});
314 torch::Tensor SolG_HV = HorizontalVertical(&tx);
315
316 // 2. ------ Case where the solutions are parallel
317 torch::Tensor g22x = G.index({g22__});
318 torch::Tensor g22f = CoF.index({g22__});
319 torch::Tensor SolG_Para = Parallel(&g22x, &g22f);
320
321 // 3. ------- Case where the solutions are intersecting
322 g22x = G.index({g22_});
323 g22f = CoF.index({g22_});
324 tx = g22.index({g22_});
325 torch::Tensor SolG_Int = Intersecting(&g22x, &tx, &g22f);
326
327 _out.index_put_({G00_G11}, SolG_HV);
328 _out.index_put_({g22__ }, SolG_Para);
329 _out.index_put_({g22_ }, SolG_Int);
330 _out.index_put_({detA }, 0);
331
332 // Swap the XY if they were swapped previously
333 _tmp = torch::cat({
334 _out.index({swp, torch::indexing::Slice(), 1}).view(dim31),
335 _out.index({swp, torch::indexing::Slice(), 0}).view(dim31),
336 _out.index({swp, torch::indexing::Slice(), 2}).view(dim31)}, 2);
337 _out.index_put_({swp}, _tmp);
338
339 // ------ Intersection of line with Ellipse ------ //
340 torch::Tensor _t, d1, V, V_, diag;
341 V = torch::cross(_out.view(dim313), _A.view(dim133), 3);
342 V = torch::transpose(V, 2, 3);
343 V = std::get<1>(torch::linalg::eig(V));
344 V = torch::real(V);
345 V = torch::transpose(V, 2, 3);
346
347 _t = V / (V.index({
348 torch::indexing::Slice(),
349 torch::indexing::Slice(),
350 torch::indexing::Slice(),
351 2}).view(dim331));
352
353 d1 = torch::sum(((_out.view(dim313))*V), 3).pow(2);
354 V_ = torch::reshape(V, dim333);
355
356 // ------- Neutrino Solution ------ //
357 _t = _t.view({dim_i, -1, 3});
358 _t = torch::nan_to_num(_t, 0, 0);
359
360 diag = torch::matmul(V_, _A.view(dim133));
361 diag = torch::sum((diag * V_), {-1}).pow(2);
362 diag = (d1 + diag).view({dim_i, -1});
363
364 // ------- Remove false solutions ------ //
365 torch::Tensor t0_, t1_, t2_;
366 t0_ = _t.index({torch::indexing::Slice(), torch::indexing::Slice(), 0}).view({dim_i, -1});
367 t1_ = _t.index({torch::indexing::Slice(), torch::indexing::Slice(), 1}).view({dim_i, -1});
368 t2_ = _t.index({torch::indexing::Slice(), torch::indexing::Slice(), 2}).view({dim_i, -1});
369 detB = ((t0_.view({dim_i, -1, 1}) == 0) * (t1_.view({dim_i, -1, 1}) == 0)).view({dim_i, -1});
370
371 t2_.index_put_({detB}, 0);
372 diag.index_put_({detA}, -1);
373 diag.index_put_({detB}, -1);
374 diag.index_put_({diag > null}, -1);
375
376 // ------- Do the Sorting ------ //
377 torch::Tensor id = std::get<1>(diag.sort(-1, false));
378 t0_ = torch::gather(t0_, -1, id).view({dim_i, -1, 1});
379 t1_ = torch::gather(t1_, -1, id).view({dim_i, -1, 1});
380 t2_ = torch::gather(t2_, -1, id).view({dim_i, -1, 1});
381 _t = torch::cat({t0_, t1_, t2_}, -1);
382 diag = torch::gather(diag, 1, id);
383
384 int max_len = diag.size(-1) - torch::max((diag != -1).sum(-1)).item<int>();
385 _t = _t.index({
386 torch::indexing::Slice(),
387 torch::indexing::Slice(max_len, torch::indexing::None),
388 torch::indexing::Slice()});
389
390 diag = diag.index({
391 torch::indexing::Slice(),
392 torch::indexing::Slice(max_len, torch::indexing::None)});
393
394 return {_t, diag};
395}

397std::map<std::string, torch::Tensor> _xNu(
398 torch::Tensor* pmc_b , torch::Tensor* pmc_mu, torch::Tensor* met_xy,
399 torch::Tensor* masses, torch::Tensor* sigma
400){
401
402 // Base matrix - the Analytical solutions and constants
403 torch::Tensor base = nusol_::BaseMatrix(pmc_b, pmc_mu, masses);
404 torch::Tensor H_base = Rotation(pmc_b, pmc_mu, &base);
405 torch::Tensor Pi2 = Pi_2(&base);
406 torch::Tensor Derivative = operators_::Rz(&Pi2);
407 Derivative = Derivative.matmul(Shape(&Pi2, {1, 1, 0}));
408
409 // Invert the input sigma matrix
410 torch::Tensor _sigma = Sigma(met_xy, sigma);
411
412 // ------- Convert missing energy matrix into tensor ----- //
413 torch::Tensor _met_xy = _met(met_xy);
414
415 // ------- create ellipse ------- //
416 torch::Tensor dNu, X_, M_;
417 dNu = _met_xy - H_base;
418 X_ = torch::matmul(torch::transpose(dNu, 1, 2), _sigma);
419 X_ = torch::matmul(X_, dNu).view({-1, 3, 3});
420 M_ = X_.matmul(Derivative);
421 M_ = M_ + torch::transpose(M_, 1, 2);
422
423 std::map<std::string, torch::Tensor> output;
424 output["M"] = M_;
425 output["H"] = H_base;
426 output["X"] = X_;
427 return output;
428}

430std::map<std::string, torch::Tensor> nusol_::Nu(
431 torch::Tensor* pmc_b , torch::Tensor* pmc_mu, torch::Tensor* met_xy,
432 torch::Tensor* masses, torch::Tensor* sigma, double null)
433{
434 torch::Tensor v, chi2;
435 std::map<std::string, torch::Tensor> _Nu, output;
436 std::tuple<torch::Tensor, torch::Tensor> sols;
437 const unsigned int dim_i = met_xy -> size(0);
438
439 _Nu = _xNu(pmc_b, pmc_mu, met_xy, masses, sigma);
440 torch::Tensor M = _Nu["M"];
441 torch::Tensor H = _Nu["H"];
442 torch::Tensor X = _Nu["X"];
443
444 torch::Tensor circl = Shape(met_xy, {1, 1, -1});
445 sols = nusol_::Intersection(&M, &circl, null);
446 v = std::get<0>(sols).view({dim_i, -1, 1, 3});
447
448 torch::Tensor diag = std::get<1>(sols);
449 chi2 = (v * X.view({dim_i, 1, 3, 3})).sum({-1});
450 chi2 = chi2.view({dim_i, -1, 3})*v.view({dim_i, -1, 3});
451 chi2 = chi2.sum(-1);
452
453 v = (H.view({dim_i, 1, 3, 3})*v).sum(-1);
454
455 torch::Tensor t0_, t1_, t2_;
456 t0_ = v.index({torch::indexing::Slice(), torch::indexing::Slice(), 0}).view({dim_i, -1});
457 t1_ = v.index({torch::indexing::Slice(), torch::indexing::Slice(), 1}).view({dim_i, -1});
458 t2_ = v.index({torch::indexing::Slice(), torch::indexing::Slice(), 2}).view({dim_i, -1});
459
460 // ------- Do the Sorting ------ //
461 torch::Tensor id = std::get<1>(chi2.sort(-1, false));
462 t0_ = torch::gather(t0_, -1, id).view({dim_i, -1, 1});
463 t1_ = torch::gather(t1_, -1, id).view({dim_i, -1, 1});
464 t2_ = torch::gather(t2_, -1, id).view({dim_i, -1, 1});
465 v = torch::cat({t0_, t1_, t2_}, -1);
466 chi2 = torch::gather(diag, 1, id);
467
468 output["nu1"] = v;
469 output["distance"] = chi2;
470 return output;
471}

478 torch::Tensor H1, H2, H1_perp, H2_perp, N1, N2, S, n_, none;

488 H2 = nusol_::BaseMatrix(pmc_b2, pmc_l2, m2);

496

499 output["nu_1"] = torch::zeros({1, 3}, op);

501 output["H_perp_1"] = H1;
502 output["H_perp_2"] = H2;
503 return output;
504 }
505
506 torch::Tensor xb1_ = pmc_b1 -> index({none});
507 torch::Tensor xl1_ = pmc_l1 -> index({none});
508 torch::Tensor h1_ = H1.index({none});
509 H1 = Rotation(&xb1_, &xl1_, &h1_);
510
511 torch::Tensor xb2_ = pmc_b2 -> index({none});
512 torch::Tensor xl2_ = pmc_l2 -> index({none});

517 H2_perp = _H_perp(&H2);
518

520 N2 = _N(&H2_perp);
521
522 S = met_xy -> index({none});
523 S = _met(&S) - Shape(&S, {1, 1, -1});
524 n_ = torch::matmul(torch::matmul(S.transpose(1, 2), N2), S);
525
526 // ---------------- Start Algorithm ----------------- //
527 sols = nusol_::Intersection(&N1, &n_, null);
528 torch::Tensor d = std::get<1>(sols);
529 torch::Tensor v = std::get<0>(sols).view({dim_i, -1, 1, 3});
530 torch::Tensor v_ = torch::sum(S.view({dim_i, 1, 3, 3}) * v, -1).view({dim_i, -1, 1, 3});
531
532 torch::Tensor K1 = torch::matmul(H1, torch::inverse(H1_perp)).view({dim_i, 1, 3, 3});
533 torch::Tensor K2 = torch::matmul(H2, torch::inverse(H2_perp)).view({dim_i, 1, 3, 3});
534
535 v = (K1 * v).sum(-1);
536 v_ = (K2 * v_).sum(-1);
537
538 unsigned int dim_j = v.size(1);
539 unsigned int dim_k = v.size(2);
540 unsigned int dim_i_ = none.size(0);
541
542 zero = torch::zeros({dim_i_, dim_j, dim_k}, op);
543 _d = torch::zeros({dim_i_, dim_j}, op);
544 _d.index_put_({none}, d);
545
546 _v = zero.clone();
547 _v.index_put_({none}, v);
548
549 __v = zero.clone();
550 __v.index_put_({none}, v_);
551
552 output["nu_1"] = _v;
553 output["nu_2"] = __v;
554 output["distance"] = _d;
555
556 output["n_"] = n_;
557 output["H_perp_1"] = H1_perp;
558 output["H_perp_2"] = H2_perp;
559 return output;
560}
