12 * @brief Counts the number of events and particle IDs in a batch.
13 *
14 * @tparam scalar_t The data type of the tensor elements.
15 * @param batch Tensor containing batch indices.
16 * @param pid Tensor containing particle IDs.
17 * @param num_events Tensor to store the number of events.
18 * @param num_pid Tensor to store the number of particle IDs.
19 * @param dx The size of the batch.
20 */
21template <typename scalar_t>
22__global__ void _count(
23 const torch::PackedTensorAccessor64<scalar_t, 1, torch::RestrictPtrTraits> batch,
24 const torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> pid,
25 torch::PackedTensorAccessor64<scalar_t, 1, torch::RestrictPtrTraits> num_events,
26 torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> num_pid,
27 const unsigned int dx
28){
29 extern __shared__ long _batch_idx[];
30 _batch_idx[threadIdx.x] = batch[threadIdx.x];
31 _batch_idx[threadIdx.x + dx] = pid[threadIdx.x][0];
32 _batch_idx[threadIdx.x + 2*dx] = pid[threadIdx.x][1];
33 __syncthreads();
34 long id = _batch_idx[threadIdx.x];
35
36 long ev = 0;
37 long sx = -1;
38 long num_lep = 0;
39 long num_bqk = 0;
40 for (size_t x(0); x < dx; ++x){
41 if (!(_batch_idx[x] == id)){continue;}
42 num_lep += (_batch_idx[x + dx ] == 1);
43 num_bqk += (_batch_idx[x + dx*2] == 1);
44 sx = (sx < 0) ? x : sx;
45 ++ev;
46 }
47 if (threadIdx.x != sx){return;}
48 num_events[id] = ev;
49 num_pid[id][0] = num_lep;
50 num_pid[id][1] = num_bqk;
51}

53/**
54 * @brief Generates combinations of leptons and b-quarks for events.
55 *
56 * @tparam scalar_t The data type of the tensor elements.
57 * @param pid Tensor containing particle IDs.
58 * @param batch Tensor containing batch indices.
59 * @param num_edges Tensor containing the number of edges.
60 * @param edge_idx Tensor containing edge indices.
61 * @param num_pid Tensor containing the number of particle IDs.
62 * @param llbb Tensor to store the combinations of leptons and b-quarks.
63 * @param msk Tensor to store the mask for valid combinations.
64 * @param n_edges The number of edges.
65 * @param mx The maximum number of combinations.
66 */
67template <typename scalar_t>
68__global__ void _combination(
69 const torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> pid,
70 const torch::PackedTensorAccessor64<scalar_t, 1, torch::RestrictPtrTraits> batch,
71 const torch::PackedTensorAccessor64<scalar_t, 1, torch::RestrictPtrTraits> num_edges,
72 const torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> edge_idx,
73 const torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> num_pid,
74 torch::PackedTensorAccessor64<scalar_t, 2, torch::RestrictPtrTraits> llbb,
75 torch::PackedTensorAccessor64<scalar_t, 1, torch::RestrictPtrTraits> msk,
76 const unsigned int n_edges, const unsigned mx
77){
78 const unsigned int __idx = blockIdx.x * blockDim.x + threadIdx.x;
79
80 const unsigned int _idx = __idx / mx;
81 long src_i = edge_idx[0][_idx];
82 long dst_i = edge_idx[1][_idx];
83 long ev_dx = batch[src_i];
84
85 const unsigned int _idy = num_edges[ev_dx] + __idx % mx;
86 if (n_edges <= _idy){return;}
87
88 long src_j = edge_idx[0][_idy];
89 long dst_j = edge_idx[1][_idy];
90 long ev_dy = batch[src_j];
91 if (ev_dx != ev_dy){return;}
92
93 long num_l = num_pid[ev_dx][0];
94 long num_b = num_pid[ev_dx][1];
95 if (num_l > 2 || num_b < 1 || num_l == 0){return;}
96
97 bool l_src_i = pid[src_i][0] > 0;
98 bool l_dst_i = pid[dst_i][0] > 0;
99 bool l_src_j = pid[src_j][0] > 0;
100 bool l_dst_j = pid[dst_j][0] > 0;
101
102 bool b_src_i = pid[src_i][1] > 0;
103 bool b_dst_i = pid[dst_i][1] > 0;
104 bool b_src_j = pid[src_j][1] > 0;
105 bool b_dst_j = pid[dst_j][1] > 0;
106
107
108 // both source and dest are leptons
109 int ll = (src_i != src_j)*(l_src_i * l_src_j);
110
111 // both source and dest are b-quarks
112 int bb = (dst_i != dst_j)*(b_dst_i * b_dst_j);
113
114 // source is lep and dest is b-quark
115 int lb = (src_i != dst_i)*(l_src_i * b_dst_i)*(dst_i != dst_j);
116
117 // source is lep and dest is b-quark
118 int lb_ = (src_j != dst_j)*(l_src_j * b_dst_j)*(src_i != src_j);
119
120 bool lx = ((ll + bb) == 2) + ((lb + lb_) == num_l);
121 if (!lx){return;}
122 if (((dst_i != dst_j) + (l_src_i) + (src_i != dst_i)) * (num_l == 1)){return;}
123
124 llbb[__idx][0] = (l_src_i)*src_i - 1 * (!l_src_i); // l1
125 llbb[__idx][1] = (l_src_j)*src_j - 1 * (!l_src_j); // l2
126 llbb[__idx][2] = (b_dst_i)*dst_i - 1 * (!b_dst_i); // b1
127 llbb[__idx][3] = (b_dst_j)*dst_j - 1 * (!b_dst_j); // b2
128 llbb[__idx][4] = ev_dx;
129 msk[__idx] = lx*num_l;
130}

133 * @brief Computes the mass matrix for given parameters.
134 *
135 * @param mass_ Tensor to store the mass matrix.
136 * @param mTl Top mass for leptons.
137 * @param mTs Top mass for b-quarks.
138 * @param mWl W mass for leptons.
139 * @param mWs W mass for b-quarks.
140 * @param steps The number of steps for computation.
141 */
142__global__ void _mass_matrix(
143 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> mass_,
144 double mTl, double mTs, double mWl, double mWs, unsigned int steps
145){
146 const unsigned int _idx = blockIdx.x * blockDim.x + threadIdx.x;
147 if (steps <= _idx){return;}
148 double lw = mWl*threadIdx.y + mTl*(1 - threadIdx.y);
149 double dx = mWs*threadIdx.y + mTs*(1 - threadIdx.y);
150 mass_[_idx][threadIdx.y] = lw + mass_[_idx][threadIdx.y]*dx*_idx;
151}

166template <size_t size_x>
167__global__ void _perturbation(
168 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_tw1,
169 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_tw2,
170 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_met,
171 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_res,
172 const double perturb, const double top_mass,
173 const double w_mass, const bool start
174){
175 __shared__ double _dnu_res[size_x][36][6];
176 __shared__ double _dnu_par[size_x][36][6];
177
178 const unsigned int idx = threadIdx.x;
179 const unsigned int idy = threadIdx.y;
180 const unsigned int idz = threadIdx.z;
181 const unsigned int _idx = (blockIdx.x * blockDim.x + threadIdx.x)*36 + idy;
182 if (_idx >= dnu_res.size({0})){return;}
183
184 if (idz < 2){ _dnu_par[idx][idy][idz] = dnu_tw1[_idx][idz] * top_mass;}
185 else if (idz < 4){_dnu_par[idx][idy][idz] = dnu_tw2[_idx][idz] * w_mass;}
186 else if (idz < 6){_dnu_par[idx][idy][idz] = dnu_met[_idx][idz];}
187 _dnu_res[idx][idy][idz] = dnu_res[_idx][idz];
188 __syncthreads();

211 * @brief Perturbs neutrino parameters for optimization.
212 *
213 * @tparam size_x The size of the shared memory.
214 * @param nu_params Tensor containing the neutrino parameters.
215 * @param dnu_met Tensor containing the MET parameters.
216 * @param dnu_tw1 Tensor containing the first set of neutrino parameters.
217 * @param dnu_tw2 Tensor containing the second set of neutrino parameters.
218 * @param lnx The length of the tensor.
219 * @param dt The perturbation step size.
220 * @param ofs The offset for the tensor.
221 */
222template <size_t size_x>
223__global__ void _perturb(
224 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> nu_params,
225 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_met,
226 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_tw1,
227 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_tw2,
228 const unsigned long lnx, const double dt, const unsigned int ofs
229){
230 __shared__ double _params_[size_x][6];
231
232 const unsigned int _idx = threadIdx.x;
233 const unsigned int _idy = threadIdx.y;
234 const unsigned int _idz = threadIdx.z;
235 const unsigned int idx = blockIdx.x * blockDim.x + _idx;
236 const unsigned int _idt = idx * ofs + _idy;
237 if (idx >= lnx){return;}
238 if (!_idy){_params_[_idx][_idz] = nu_params[idx][_idz];}
239 __syncthreads();
240
241 double dx_ = _params_[_idx][_idz] + (_idy == _idz) * dt;
242 if (_idz < 2){dnu_met[_idt][_idz ] = dx_; return;}
243 if (_idz < 4){dnu_tw1[_idt][_idz-2] = dx_; return;}
244 if (_idz < 6){dnu_tw2[_idt][_idz-4] = dx_; return;}

247/**
248 * @brief Computes the Jacobian matrix for optimization.
249 *
250 * @tparam size_x The size of the shared memory.
251 * @param nu_params Tensor containing the neutrino parameters.
252 * @param dnu_res Tensor containing the result of the perturbation.
253 * @param lnx The length of the tensor.
254 * @param dt The perturbation step size.
255 * @param ofs The offset for the tensor.
256 */
257template <size_t size_x>
258__global__ void _jacobi(
259 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> nu_params,
260 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> dnu_res,
261 const unsigned long lnx, const double dt, unsigned int ofs
262){
263 __shared__ double _param_[size_x][6];
264 __shared__ double _Jxt_[size_x][6];
265
266 const unsigned int _idx = threadIdx.x;
267 const unsigned int _idy = threadIdx.y;
268 const unsigned int _idp = blockIdx.x * blockDim.x + _idx;
269 const unsigned int idx = _idp*ofs + _idy; // compute the offset (0 -> 5)
270 const unsigned int _udp = (_idp + 1)*ofs-1; // unperturbed index ((n+1)*6)
271 if (idx >= lnx*ofs){return;}
272
273 // ----- Get the (un)perturbed values ----- //
274 double v = dnu_res[_udp][0];
275 double dv = dnu_res[idx][0];
276 _param_[_idx][_idy] = nu_params[_idp][_idy];
277
278 // ----- (compute derivative) ----- //
279 dv = (dv - v) / dt;
280 _Jxt_[_idx][_idy] = dv*dv;
281 __syncthreads();
282
283 // ----- update state ------- //
284 dv = _div(_sum(_Jxt_[_idx], 6)) * dv * v;
285 nu_params[_idp][_idy] = _param_[_idx][_idy] - dv;
286}

288/**
289 * @brief Compares solutions and selects the best one.
290 *
291 * @tparam size_x The size of the shared memory.
292 * @param cmx_dx Tensor containing combination indices.
293 * @param evnt_dx Tensor containing event indices.
294 * @param o_sol Tensor to store the best solution.
295 * @param o_cmb Tensor to store the best combination.
296 * @param o_nu1 Tensor to store the first neutrino parameters.
297 * @param o_nu2 Tensor to store the second neutrino parameters.
298 * @param i_cmb Tensor containing input combinations.
299 * @param i_sol Tensor containing input solutions.
300 * @param i_nu1 Tensor containing input first neutrino parameters.
301 * @param i_nu2 Tensor containing input second neutrino parameters.
302 * @param lx The length of the tensor.
303 * @param evnts The number of events.
304 */
305template <size_t size_x>
306__global__ void _compare_solx(
307 torch::PackedTensorAccessor64<long , 1, torch::RestrictPtrTraits> cmx_dx,
308 torch::PackedTensorAccessor64<long , 1, torch::RestrictPtrTraits> evnt_dx,
309
310 torch::PackedTensorAccessor64<double, 1, torch::RestrictPtrTraits> o_sol,
311 torch::PackedTensorAccessor64<long , 2, torch::RestrictPtrTraits> o_cmb,
312 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> o_nu1,
313 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> o_nu2,
314
315 torch::PackedTensorAccessor64<long , 2, torch::RestrictPtrTraits> i_cmb,
316 torch::PackedTensorAccessor64<double, 2, torch::RestrictPtrTraits> i_sol,
317 torch::PackedTensorAccessor64<double, 3, torch::RestrictPtrTraits> i_nu1,
318 torch::PackedTensorAccessor64<double, 3, torch::RestrictPtrTraits> i_nu2,
319 const unsigned int lx, const unsigned int evnts
320){
321
322 __shared__ double _nu1[size_x][4];
323 __shared__ double _nu2[size_x][4];
324 __shared__ long _cmx[size_x][4];
325 __shared__ double _score[size_x][4];
326
327 _nu1[threadIdx.x][threadIdx.y] = 0;
328 _nu2[threadIdx.x][threadIdx.y] = 0;
329 _cmx[threadIdx.x][threadIdx.y] = 0;
330 _score[threadIdx.x][threadIdx.y] = 0;
331 const unsigned int _idx = blockIdx.x * blockDim.x + threadIdx.x;
332 if (_idx >= evnts){return;}
333
334 for (size_t x(0); x < lx; ++x){
335 long cmx_id = cmx_dx[x];
336 long evn_id = evnt_dx[cmx_id];
337 if (evn_id != _idx){continue;}
338 for (size_t y(0); y < 6; ++y){
339 double sol = i_sol[x][y];
340 if (_score[threadIdx.x][threadIdx.y] < sol){continue;}
341 _score[threadIdx.x][threadIdx.y] = sol;
342 _cmx[threadIdx.x][threadIdx.y] = i_cmb[cmx_id][threadIdx.y];
343 _nu1[threadIdx.x][threadIdx.y] = (threadIdx.y < 3) ? i_nu1[x][y][threadIdx.y] : 0;
344 _nu2[threadIdx.x][threadIdx.y] = (threadIdx.y < 3) ? i_nu2[x][y][threadIdx.y] : 0;
345 }
346 }
347
348 __syncthreads();
349 o_sol[_idx] = _score[threadIdx.x][threadIdx.y];
350 o_cmb[_idx][threadIdx.y] = _cmx[threadIdx.x][threadIdx.y];
351 o_nu1[_idx][threadIdx.y] = (threadIdx.y < 3) ? _nu1[threadIdx.x][threadIdx.y] : _sqrt(_dot(_nu1[threadIdx.x], _nu1[threadIdx.x], 3));
352 o_nu2[_idx][threadIdx.y] = (threadIdx.y < 3) ? _nu2[threadIdx.x][threadIdx.y] : _sqrt(_dot(_nu2[threadIdx.x], _nu2[threadIdx.x], 3));
353}
