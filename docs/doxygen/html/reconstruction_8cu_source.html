100 _found[_idz] = (ix_ == ix);
101
102 __syncthreads();

104 for (size_t f(0); f < num_n; ++f){found *= _found[f];}
105 __syncthreads();

107 if (!found){_found[_idz] = false; continue;}
108 trk[idx][x] = x;
109 _found[_idz] = true;

111 break;
112 }
113 if (!_found[_idz] || ix < 0 || pos < 0){return;}

120 px += pmc[id][0];
121 py += pmc[id][1];
122 pz += pmc[id][2];
123 e += pmc[id][3];
124 }
