A complete analysis example in Python.

Imports

from AnalysisG.core import Analysis
from AnalysisG.core import EventTemplate, ParticleTemplate
from AnalysisG.core import GraphTemplate
from AnalysisG.core import ModelTemplate, OptimizerConfig
from AnalysisG.core import SelectionTemplate
import torch
import torch.nn as nn

Define Particles

class Jet(ParticleTemplate):
def __init__(self):
ParticleTemplate.__init__(self)
self.Leaves = ["pt", "eta", "phi", "e", "btag"]
@property
def is_bjet(self):
return self.btag > 0.7

Define Event

class TTbarEvent(EventTemplate):
def __init__(self):
EventTemplate.__init__(self)
self.name = "ttbar"
self.Trees = ["nominal"]
self.Branches = ["jet_"]
self.Leaves = ["met_et", "met_phi"]
self.Objects = {"Jets": Jet}

Define Graph

class JetGraph(GraphTemplate):
def __init__(self):
GraphTemplate.__init__(self)
self.name = "jet_graph"
def CompileEvent(self, event):
self.DefineNodes(event.Jets)
self.AddNodeFeature("pt", lambda j: j.pt / 1000)
self.AddNodeFeature("eta", lambda j: j.eta)
self.AddNodeFeature("phi", lambda j: j.phi)
self.AddNodeTruth("is_b", lambda j: int(j.is_bjet))
def PreSelection(self, event):
return len(event.Jets) >= 4

Define Model

class JetTagger(ModelTemplate):
def __init__(self):
ModelTemplate.__init__(self)
self.fc = nn.Sequential(
nn.Linear(3, 64),
nn.ReLU(),
nn.Linear(64, 32),
nn.ReLU(),
nn.Linear(32, 1),
nn.Sigmoid()
)
self.i_node = ["pt", "eta", "phi"]
self.o_node = {"pred": "is_b"}
def forward(self, data):
x = data.node_features
return {"pred": self.fc(x)}

Run Analysis

# Create analysis
ana = Analysis()
# Configure
ana.OutputPath = "results/"
ana.Epochs = 50
ana.kFolds = 5
ana.BatchSize = 64
ana.Threads = 8
# Add data
ana.AddSamples("/data/ttbar/", "ttbar")
# Add templates
ana.AddEvent(TTbarEvent(), "ttbar")
ana.AddGraph(JetGraph(), "ttbar")
# Configure optimizer
opt = OptimizerConfig()
opt.optimizer = "Adam"
opt.lr = 0.001
# Add model
ana.AddModel(JetTagger(), opt, "btagger")
# Run
ana.Start()
