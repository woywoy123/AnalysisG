Usage example:

class MyClass {
public:
cproperty<int, MyClass> value;
MyClass() {
value.set_getter(get_value);
value.set_setter(set_value);
value.set_object(this);
}
private:
int m_value;
static void get_value(int* out, MyClass* obj) { *out = obj->m_value; }
static void set_value(int* in, MyClass* obj) { obj->m_value = *in; }
};
